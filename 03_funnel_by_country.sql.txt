-- =====================================================
-- FUNNEL SEGMENTATION BY COUNTRY
-- Compare conversion performance between countries
-- =====================================================

WITH first_visits AS (
    SELECT DISTINCT user_id, country
    FROM mercadolibre_funnel
    WHERE event_name = 'first_visit'
),

select_item AS (
    SELECT DISTINCT user_id, country
    FROM mercadolibre_funnel
    WHERE event_name IN ('select_item','select_promotion')
),

add_to_cart AS (
    SELECT DISTINCT user_id, country
    FROM mercadolibre_funnel
    WHERE event_name = 'add_to_cart'
),

purchase AS (
    SELECT DISTINCT user_id, country
    FROM mercadolibre_funnel
    WHERE event_name = 'purchase'
)

SELECT
    fv.country,
    COUNT(fv.user_id) AS users_first_visit,
    COUNT(p.user_id) AS users_purchase,

    ROUND(
        COUNT(p.user_id) * 100.0 /
        NULLIF(COUNT(fv.user_id),0),
        2
    ) AS conversion_rate

FROM first_visits fv
LEFT JOIN purchase p
ON fv.user_id = p.user_id
AND fv.country = p.country

GROUP BY fv.country
ORDER BY conversion_rate DESC;
